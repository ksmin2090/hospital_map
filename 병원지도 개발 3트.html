(특수문자가 없어서 그대로 넣어도 무방!) const SERVICE_KEY = '2016f36910df036929379edb9f10c17d0030f48c2654157a7c0e8fb6f0ad373a'; const statusEl = document.getElementById('status'); // 2. 상세 페이지에 적힌 최신 v2 주소를 반영했어! const targetUrl = `https://apis.data.go.kr/B551182/hospInfoServicev2/getHospBasisList?serviceKey=${SERVICE_KEY}&_type=json&numOfRows=100&clCd=01`; const proxyUrl = `https://api.allorigins.win/get?url=${encodeURIComponent(targetUrl)}`; try { const response = await fetch(proxyUrl); const data = await response.json(); const result = JSON.parse(data.contents); if(!result.response.body.items || !result.response.body.items.item) { statusEl.innerText = "❌ 데이터가 없습니다. ${hosp.addr}`); } }); statusEl.innerText = `✅ 성공: ${hospitalList.length}개의 종합병원을 찾았습니다!`; } catch (error) { statusEl.innerText = "❌ 키가 등록되지 않았거나 서버 오류입니다."; console.error(error); } }
